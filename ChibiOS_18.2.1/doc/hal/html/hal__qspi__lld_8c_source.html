54 /**
55  * @brief Shared service routine.
56  *
57  * @param[in] qspip pointer to the @p QSPIDriver object
58  * @param[in] flags pre-shifted content of the ISR register
59  */
60 static void qspi_lld_serve_dma_interrupt(QSPIDriver *qspip, uint32_t flags) {
61 
62  (void)qspip;
63  (void)flags;
64 
65  /* DMA errors handling.*/
66 #if defined(STM32_QSPI_DMA_ERROR_HOOK)
67  if ((flags & (STM32_DMA_ISR_TEIF | STM32_DMA_ISR_DMEIF)) != 0) {
68  STM32_QSPI_DMA_ERROR_HOOK(qspip);
69  }
70 #endif
71 }

73 /**
74  * @brief Shared service routine.
75  *
76  * @param[in] qspip pointer to the @p QSPIDriver object
77  */
78 static void qspi_lld_serve_interrupt(QSPIDriver *qspip) {
79 
80  /* Portable QSPI ISR code defined in the high level driver, note, it is
81  a macro.*/
82  _qspi_isr_code(qspip);
83 
84  /* Stop everything, we need to give DMA enough time to complete the ongoing
85  operation. Race condition hidden here.*/
86  while (dmaStreamGetTransactionSize(qspip->dma) > 0U)
87  ;
88  dmaStreamDisable(qspip->dma);
89 }

145 /**
146  * @brief Configures and activates the QSPI peripheral.
147  *
148  * @param[in] qspip pointer to the @p QSPIDriver object
149  *
150  * @notapi
151  */
152 void qspi_lld_start(QSPIDriver *qspip) {
153 
154  /* If in stopped state then full initialization.*/
155  if (qspip->state == QSPI_STOP) {
156 #if STM32_QSPI_USE_QUADSPI1
157  if (&QSPID1 == qspip) {
158  bool b = dmaStreamAllocate(qspip->dma,
159  STM32_QSPI_QUADSPI1_DMA_IRQ_PRIORITY,
160  (stm32_dmaisr_t)qspi_lld_serve_dma_interrupt,
161  (void *)qspip);
162  osalDbgAssert(!b, "stream already allocated");
163  rccEnableQUADSPI1(true);
164  }
165 #endif
166 
167  /* Common initializations.*/
168  dmaStreamSetPeripheral(qspip->dma, &qspip->qspi->DR);
169  }
170 
171  /* QSPI setup and enable.*/
172  qspip->qspi->DCR = qspip->config->dcr;
173  qspip->qspi->CR = ((STM32_QSPI_QUADSPI1_PRESCALER_VALUE - 1U) << 24U) |
174  QUADSPI_CR_TCIE | QUADSPI_CR_DMAEN | QUADSPI_CR_EN;
175  qspip->qspi->FCR = QUADSPI_FCR_CTEF | QUADSPI_FCR_CTCF |
176  QUADSPI_FCR_CSMF | QUADSPI_FCR_CTOF;
177 }

181  *
182  * @param[in] qspip pointer to the @p QSPIDriver object
183  *
184  * @notapi
185  */
186 void qspi_lld_stop(QSPIDriver *qspip) {
187 
188  /* If in ready state then disables the QUADSPI clock.*/
189  if (qspip->state == QSPI_READY) {
190 
191  /* QSPI disable.*/
192  qspip->qspi->CR = 0U;
193 
194  /* Releasing the DMA.*/
195  dmaStreamRelease(qspip->dma);
196 
197  /* Stopping involved clocks.*/
198 #if STM32_QSPI_USE_QUADSPI1
199  if (&QSPID1 == qspip) {
200  rccDisableQUADSPI1();
201  }
202 #endif
203  }

206 /**
207  * @brief Sends a command without data phase.
208  * @post At the end of the operation the configured callback is invoked.
209  *
210  * @param[in] qspip pointer to the @p QSPIDriver object
211  * @param[in] cmd pointer to the command descriptor
212  *
213  * @notapi
214  */
215 void qspi_lld_command(QSPIDriver *qspip, const qspi_command_t *cmdp) {
216 
217 #if STM32_USE_STM32_D1_WORKAROUND == TRUE
218  /* If it is a command without address and alternate phases then the command
219  is sent as an alternate byte, the command phase is suppressed.*/
220  if ((cmdp->cfg & (QSPI_CFG_ADDR_MODE_MASK | QSPI_CFG_ALT_MODE_MASK)) == 0U) {
221  uint32_t cfg;
222 
223  /* The command mode field is copied in the alternate mode field. All
224  other fields are not used in this scenario.*/
225  cfg = (cmdp->cfg & QSPI_CFG_CMD_MODE_MASK) << 6U;
226 
227  qspip->qspi->DLR = 0U;
228  qspip->qspi->ABR = cmdp->cfg & QSPI_CFG_CMD_MASK;
229  qspip->qspi->CCR = cfg;
230  return;
231  }
232 #endif
233  qspip->qspi->DLR = 0U;
234  qspip->qspi->ABR = cmdp->alt;
235  qspip->qspi->CCR = cmdp->cfg;
236  if ((cmdp->cfg & QSPI_CFG_ADDR_MODE_MASK) != QSPI_CFG_ADDR_MODE_NONE) {
237  qspip->qspi->AR = cmdp->addr;
238  }
239 }

241 /**
242  * @brief Sends a command with data over the QSPI bus.
243  * @post At the end of the operation the configured callback is invoked.
244  *
245  * @param[in] qspip pointer to the @p QSPIDriver object
246  * @param[in] cmd pointer to the command descriptor
247  * @param[in] n number of bytes to send
248  * @param[in] txbuf the pointer to the transmit buffer
249  *
250  * @notapi
251  */
252 void qspi_lld_send(QSPIDriver *qspip, const qspi_command_t *cmdp,
253  size_t n, const uint8_t *txbuf) {
254 
255  dmaStreamSetMemory0(qspip->dma, txbuf);
256  dmaStreamSetTransactionSize(qspip->dma, n);
257  dmaStreamSetMode(qspip->dma, qspip->dmamode | STM32_DMA_CR_DIR_M2P);
258 
259  qspip->qspi->DLR = n - 1;
260  qspip->qspi->ABR = cmdp->alt;
261  qspip->qspi->CCR = cmdp->cfg;
262  if ((cmdp->cfg & QSPI_CFG_ADDR_MODE_MASK) != QSPI_CFG_ADDR_MODE_NONE) {
263  qspip->qspi->AR = cmdp->addr;
264  }
265 
266  dmaStreamEnable(qspip->dma);
267 }

269 /**
270  * @brief Sends a command then receives data over the QSPI bus.
271  * @post At the end of the operation the configured callback is invoked.
272  *
273  * @param[in] qspip pointer to the @p QSPIDriver object
274  * @param[in] cmd pointer to the command descriptor
275  * @param[in] n number of bytes to send
276  * @param[out] rxbuf the pointer to the receive buffer
277  *
278  * @notapi
279  */
280 void qspi_lld_receive(QSPIDriver *qspip, const qspi_command_t *cmdp,
281  size_t n, uint8_t *rxbuf) {
282 
283  dmaStreamSetMemory0(qspip->dma, rxbuf);
284  dmaStreamSetTransactionSize(qspip->dma, n);
285  dmaStreamSetMode(qspip->dma, qspip->dmamode | STM32_DMA_CR_DIR_P2M);
286 
287  qspip->qspi->DLR = n - 1;
288  qspip->qspi->ABR = cmdp->alt;
289  qspip->qspi->CCR = cmdp->cfg | QUADSPI_CCR_FMODE_0;
290  if ((cmdp->cfg & QSPI_CFG_ADDR_MODE_MASK) != QSPI_CFG_ADDR_MODE_NONE) {
291  qspip->qspi->AR = cmdp->addr;
292  }
293 
294  dmaStreamEnable(qspip->dma);
295 }

298 /**
299  * @brief Maps in memory space a QSPI flash device.
300  * @pre The memory flash device must be initialized appropriately
301  * before mapping it in memory space.
302  *
303  * @param[in] qspip pointer to the @p QSPIDriver object
304  * @param[in] cmdp pointer to the command descriptor
305  * @param[out] addrp pointer to the memory start address of the mapped
306  * flash or @p NULL
307  *
308  * @notapi
309  */
310 void qspi_lld_map_flash(QSPIDriver *qspip,
311  const qspi_command_t *cmdp,
312  uint8_t **addrp) {
313 
314  /* Disabling the DMA request while in memory mapped mode.*/
315  qspip->qspi->CR &= ~QUADSPI_CR_DMAEN;
316 
317  /* Starting memory mapped mode using the passed parameters.*/
318  qspip->qspi->DLR = 0;
319  qspip->qspi->ABR = 0;
320  qspip->qspi->AR = 0;
321  qspip->qspi->CCR = cmdp->cfg | QUADSPI_CCR_FMODE_1 | QUADSPI_CCR_FMODE_0;
322 
323  /* Mapped flash absolute base address.*/
324  if (addrp != NULL) {
325  *addrp = (uint8_t *)0x90000000;
326  }
327 }

329 /**
330  * @brief Unmaps from memory space a QSPI flash device.
331  * @post The memory flash device must be re-initialized for normal
332  * commands exchange.
333  *
334  * @param[in] qspip pointer to the @p QSPIDriver object
335  *
336  * @notapi
337  */
338 void qspi_lld_unmap_flash(QSPIDriver *qspip) {
339 
340  /* Aborting memory mapped mode.*/
341  qspip->qspi->CR |= QUADSPI_CR_ABORT;
342  while ((qspip->qspi->CR & QUADSPI_CR_ABORT) != 0U) {
343  }
344 
345  /* Re-enabling DMA request, we are going back to indirect mode.*/
346  qspip->qspi->CR |= QUADSPI_CR_DMAEN;
347 }
