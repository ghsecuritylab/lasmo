74 /**
75  * @brief Pads mode setup.
76  * @details This function programs a pads group belonging to the same port
77  * with the specified mode.
78  * @note @p PAL_MODE_UNCONNECTED is implemented as push pull at minimum
79  * speed.
80  *
81  * @param[in] port the port identifier
82  * @param[in] mask the group mask
83  * @param[in] mode the mode
84  *
85  * @notapi
86  */
87 void _pal_lld_setgroupmode(ioportid_t port,
88  ioportmask_t mask,
89  iomode_t mode) {
90 
91  uint32_t moder = (mode & PAL_STM32_MODE_MASK) >> 0;
92  uint32_t otyper = (mode & PAL_STM32_OTYPE_MASK) >> 2;
93  uint32_t ospeedr = (mode & PAL_STM32_OSPEED_MASK) >> 3;
94  uint32_t pupdr = (mode & PAL_STM32_PUPDR_MASK) >> 5;
95  uint32_t altr = (mode & PAL_STM32_ALTERNATE_MASK) >> 7;
96  uint32_t bit = 0;
97  while (true) {
98  if ((mask & 1) != 0) {
99  uint32_t altrmask, m1, m2, m4;
100 
101  altrmask = altr << ((bit & 7) * 4);
102  m1 = 1 << bit;
103  m2 = 3 << (bit * 2);
104  m4 = 15 << ((bit & 7) * 4);
105  port->OTYPER = (port->OTYPER & ~m1) | otyper;
106  port->OSPEEDR = (port->OSPEEDR & ~m2) | ospeedr;
107  port->PUPDR = (port->PUPDR & ~m2) | pupdr;
108  if ((mode & PAL_STM32_MODE_MASK) == PAL_STM32_MODE_ALTERNATE) {
109  /* If going in alternate mode then the alternate number is set
110  before switching mode in order to avoid glitches.*/
111  if (bit < 8)
112  port->AFRL = (port->AFRL & ~m4) | altrmask;
113  else
114  port->AFRH = (port->AFRH & ~m4) | altrmask;
115  port->MODER = (port->MODER & ~m2) | moder;
116  }
117  else {
118  /* If going into a non-alternate mode then the mode is switched
119  before setting the alternate mode in order to avoid glitches.*/
120  port->MODER = (port->MODER & ~m2) | moder;
121  if (bit < 8)
122  port->AFRL = (port->AFRL & ~m4) | altrmask;
123  else
124  port->AFRH = (port->AFRH & ~m4) | altrmask;
125  }
126  }
127  mask >>= 1;
128  if (!mask)
129  return;
130  otyper <<= 1;
131  ospeedr <<= 2;
132  pupdr <<= 2;
133  moder <<= 2;
134  bit++;
135  }
136 }

139 /**
140  * @brief Pad event enable.
141  * @note Programming an unknown or unsupported mode is silently ignored.
142  *
143  * @param[in] port port identifier
144  * @param[in] pad pad number within the port
145  * @param[in] mode pad event mode
146  *
147  * @notapi
148  */
149 void _pal_lld_enablepadevent(ioportid_t port,
150  iopadid_t pad,
151  ioeventmode_t mode) {
152 
153  uint32_t padmask, cridx, croff, crmask, portidx;
154 
155  /* Mask of the pad.*/
156  padmask = 1U << (uint32_t)pad;
157 
158  /* Multiple channel setting of the same channel not allowed, first disable
159  it. This is done because on STM32 the same channel cannot be mapped on
160  multiple ports.*/
161 #if defined(STM32_EXTI_ENHANCED)
162  osalDbgAssert(((EXTI->RTSR1 & padmask) == 0U) &&
163  ((EXTI->FTSR1 & padmask) == 0U), "channel already in use");
164 #else
165  osalDbgAssert(((EXTI->RTSR & padmask) == 0U) &&
166  ((EXTI->FTSR & padmask) == 0U), "channel already in use");
167 #endif
168 
169  /* Index and mask of the SYSCFG CR register to be used.*/
170  cridx = (uint32_t)pad >> 2U;
171  croff = ((uint32_t)pad & 3U) * 4U;
172  crmask = ~(0xFU << croff);
173 
174  /* Port index is obtained assuming that GPIO ports are placed at regular
175  0x400 intervals in memory space. So far this is true for all devices.*/
176  portidx = (((uint32_t)port - (uint32_t)GPIOA) >> 10U) & 0xFU;
177 
178  /* Port selection in SYSCFG.*/
179  SYSCFG->EXTICR[cridx] = (SYSCFG->EXTICR[cridx] & crmask) | (portidx << croff);
180 
181  /* Programming edge registers.*/
182 #if defined(STM32_EXTI_ENHANCED)
183  if (mode & PAL_EVENT_MODE_RISING_EDGE)
184  EXTI->RTSR1 |= padmask;
185  else
186  EXTI->RTSR1 &= ~padmask;
187  if (mode & PAL_EVENT_MODE_FALLING_EDGE)
188  EXTI->FTSR1 |= padmask;
189  else
190  EXTI->FTSR1 &= ~padmask;
191 
192  /* Programming interrupt and event registers.*/
193  EXTI_D1->IMR1 |= padmask;
194  EXTI_D1->EMR1 &= ~padmask;
195 #else
196  if (mode & PAL_EVENT_MODE_RISING_EDGE)
197  EXTI->RTSR |= padmask;
198  else
199  EXTI->RTSR &= ~padmask;
200  if (mode & PAL_EVENT_MODE_FALLING_EDGE)
201  EXTI->FTSR |= padmask;
202  else
203  EXTI->FTSR &= ~padmask;
204 
205  /* Programming interrupt and event registers.*/
206  EXTI->IMR |= padmask;
207  EXTI->EMR &= ~padmask;
208 #endif
209 }

211 /**
212  * @brief Pad event disable.
213  * @details This function disables previously programmed event callbacks.
214  *
215  * @param[in] port port identifier
216  * @param[in] pad pad number within the port
217  *
218  * @notapi
219  */
220 void _pal_lld_disablepadevent(ioportid_t port, iopadid_t pad) {
221  uint32_t padmask, rtsr1, ftsr1;
222 
223 #if defined(STM32_EXTI_ENHANCED)
224  rtsr1 = EXTI->RTSR1;
225  ftsr1 = EXTI->FTSR1;
226 #else
227  rtsr1 = EXTI->RTSR;
228  ftsr1 = EXTI->FTSR;
229 #endif
230 
231  /* Mask of the pad.*/
232  padmask = 1U << (uint32_t)pad;
233 
234  /* If either RTRS1 or FTSR1 is enabled then the channel is in use.*/
235  if (((rtsr1 | ftsr1) & padmask) != 0U) {
236  uint32_t cridx, croff, crport, portidx;
237 
238  /* Index and mask of the SYSCFG CR register to be used.*/
239  cridx = (uint32_t)pad >> 2U;
240  croff = ((uint32_t)pad & 3U) * 4U;
241 
242  /* Port index is obtained assuming that GPIO ports are placed at regular
243  0x400 intervals in memory space. So far this is true for all devices.*/
244  portidx = (((uint32_t)port - (uint32_t)GPIOA) >> 10U) & 0xFU;
245 
246  crport = (SYSCFG->EXTICR[cridx] >> croff) & 0xFU;
247 
248  osalDbgAssert(crport == portidx, "channel mapped on different port");
249 
250 #if defined(STM32_EXTI_ENHANCED)
251  /* Disabling channel.*/
252  EXTI_D1->IMR1 &= ~padmask;
253  EXTI_D1->EMR1 &= ~padmask;
254  EXTI->RTSR1 = rtsr1 & ~padmask;
255  EXTI->FTSR1 = ftsr1 & ~padmask;
256  EXTI_D1->PR1 = padmask;
257 #else
258  /* Disabling channel.*/
259  EXTI->IMR &= ~padmask;
260  EXTI->EMR &= ~padmask;
261  EXTI->RTSR = rtsr1 & ~padmask;
262  EXTI->FTSR = ftsr1 & ~padmask;
263  EXTI->PR = padmask;
264 #endif
265 
266 #if PAL_USE_CALLBACKS || PAL_USE_WAIT
267  /* Callback cleared and/or thread reset.*/
268  _pal_clear_event(pad);
269 #endif
270  }
271 }
